
add_namespace = bs_ger

country_event = { #Alpes to Italy Event
	id = bs_ger.1
	title = bs_ger.1.t
	desc = bs_ger.1.d
	fire_only_once = yes
	mean_time_to_happen = { days = 3 }
	picture = GFX_report_event_german_troops
	
	trigger = {
		TAG = GER
		GER = { NOT = { has_government = democratic } }
	    735 = { is_owned_by = GER }
		837 = { is_owned_by = GER }
		838 = { is_owned_by = GER }
		country_exists = ITA
		NOT = {
		    has_global_flag = germany_ceded_alpes
		}
	}	
	
	option = { #give alpes to Italy
		name = bs_ger.1.a
		ai_chance = {
			factor = 50
		}
        
		ITA = {
		    country_event = { id = bs_ger.2 days = 0 }
			effect_tooltip = {
				transfer_state = 735
				transfer_state = 837
				transfer_state = 838
			}
		}
	}
	option = { #alpes is mine
		name = bs_ger.1.b
		ai_chance = {
			factor = 5
		}

		hidden_effect = {
		    set_global_flag = germany_ceded_alpes
		}
	}
}
